\usepackage{pgfplots}
\pgfplotsset{compat=1.15}
\usetikzlibrary{arrows,arrows.meta}

\begin{document}

\definecolor{ggbA}{RGB}{77,77,255}
\definecolor{ggbB}{RGB}{125,125,255}
\definecolor{ggbC}{RGB}{153,51,0}

\begin{tikzpicture}[scale=1]
  % 坐标点定义
  \coordinate (A) at (-1.88,2.39);
  \coordinate (B) at (1.3,1.85);
  \coordinate (C) at (0.14,0);
  \coordinate (D) at (-2.22,-0.49);

  % 几何元素
  \draw[help lines,step=1,gray!30] (-7.12,-6.3) grid (7.12,6.3);
  \draw[->,line width=0.35pt] (-7.12,0) -- (7.12,0) node[right,font=\fontsize{10pt}{12pt}\selectfont] {$x$};
  \draw[->,line width=0.35pt] (0,-6.3) -- (0,6.3) node[above,font=\fontsize{10pt}{12pt}\selectfont] {$y$};
  \fill[ggbC,opacity=0.1] (A) -- (B) -- (C) -- (D) -- cycle;
  \draw[color=ggbC,line cap=round,line join=round,line width=0.5pt] (A) -- (B) -- (C) -- (D) -- cycle;

  % 点标记
  \draw[fill=ggbA] (A) circle (1pt);
  \draw[fill=ggbA] (B) circle (1pt);
  \draw[fill=ggbB] (C) circle (1pt);
  \draw[fill=ggbA] (D) circle (1pt);

  % 标签
  \node [above left,font=\fontsize{10pt}{12pt}\selectfont] at (A) {$A$};
  \node [above right,font=\fontsize{10pt}{12pt}\selectfont] at (B) {$B$};
  \node [below right,font=\fontsize{10pt}{12pt}\selectfont] at (C) {$C$};
  \node [below left,font=\fontsize{10pt}{12pt}\selectfont] at (D) {$D$};
\end{tikzpicture}

\end{document}
